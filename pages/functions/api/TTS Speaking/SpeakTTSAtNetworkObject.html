Note

Each string[] item will be spoken sequentially.

Note

The textToSpeak input will automatically be split by sentence under the hood.

Best practice example

C# 
                
                    // If you have repeated sentences with custom inputs, then it's best to split the sentence in multiple items
                    // This way only the new word is generated and the other parts are reused from the cache
                    // See below for an example

                    // In this example I'm just gonna use a random enemy name as the input for simplicity sake
                    string[] enemyNames = new[]
                    {
                        "Barber",
                        "Bracken",
                        "Bunker Spider",
                        "Coil-Head",
                        "Hoarding Bug",
                        "Hygrodere",
                        "Jester",
                        "Nutcracker"
                    };

                    // The sentence that is getting reused
                    string[] customStrings = new[]
                    {
                        "Warning, ",
                        "ENTITYNAME",
                        " detected near your position."
                    };

                    // Get a random index
                    randomIndex = UnityEngine.Random.Range(0, enemyNames.Length);
                    // Replace "ENTITYNAME" with the random enemy name
                    customStrings[1] = enemyNames[randomIndex];

                    // Player is going to speak now
                    PlayerControllerB localPlayer = StartOfRound.Instance.localPlayerController;
                    // Now speak
                    TTSCompanyAPI.SpeakTTSAtNetworkObject(localPlayer.gameObject, customStrings);
